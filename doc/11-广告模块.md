# 租房避雷App - 广告模块设计

## 1. 模块概述

**模块名称**：广告模块 (Advertisement Module)

**功能描述**：
- 广告管理（添加、编辑、删除）
- 广告投放配置

### 4.1 计费方式

|---------|------|------|
| CPM（展示计费） | ¥0.05/千次展示 | 按展示次数计费 |
| CPC（点击计费） | ¥0.10/次点击 | 按点击次数计费 |

### 4.2 成本计算

```typescript
// 计算展示成本
function calculateCpmCost(impressions: number): number {
  return Math.ceil(impressions / 1000) * 0.05;
}

// 计算点击成本
function calculateCpcCost(clicks: number): number {
  return clicks * 0.10;
}

// 计算总成本
function calculateTotalCost(ads: Advertisement[]): number {
  let totalCost = 0;

  for (const ad of ads) {
    const cpmCost = calculateCpmCost(ad.impressions);
    const cpcCost = calculateCpcCost(ad.clicks);
    totalCost += cpmCost + cpcCost;
  }

  return totalCost;
}
```

**核心代码示例**：

```typescript
@Injectable()
export class AdvertisementService {
  constructor(
    private prisma: PrismaService,
    private redisService: RedisService,
  ) {}

  async create(dto: CreateAdvertisementDto): Promise<Advertisement> {
    // 设置默认投放时间
    const startDate = dto.startDate
      ? new Date(dto.startDate)
      : new Date();

    const endDate = dto.endDate
      ? new Date(dto.endDate)
      : new Date(Date.now() + 30 * 24 * 60 * 60 * 1000); // 30天后

    // 验证图片
    await this.validateImage(dto.image);

    return this.prisma.advertisement.create({
      data: {
        ...dto,
        status: 'active',
        impressions: 0,
        clicks: 0,
        startDate,
        endDate,
      },
    });
  }

  async getAvailableAds(position: string): Promise<Advertisement[]> {
    const now = new Date();

    // 先查缓存
    const cacheKey = `ads:available:${position}`;
    const cached = await this.redisService.get(cacheKey);
    if (cached) {
      return JSON.parse(cached);
    }

    // 查询可展示的广告
    const ads = await this.prisma.advertisement.findMany({
      where: {
        position,
        status: 'active',
        startDate: { lte: now },
        endDate: { gte: now },
      },
      orderBy: { createdAt: 'asc' },
    });

    // 缓存5分钟
    await this.redisService.set(cacheKey, JSON.stringify(ads), 300);
    return ads;
  }

  async recordImpression(id: bigint): Promise<void> {
    // 使用 Redis 原子操作避免并发问题
    const key = `ad:impression:${id}`;
    await this.redisService.incr(key);

    // 每100次展示同步到数据库
    const count = await this.redisService.get(key);
    if (parseInt(count) % 100 === 0) {
      await this.prisma.advertisement.update({
        where: { id },
        data: {
          impressions: { increment: parseInt(count) },
        },
      });
      await this.redisService.set(key, '0');
    }
  }

  async recordClick(id: bigint): Promise<void> {
    // 使用 Redis 原子操作
    const key = `ad:click:${id}`;
    await this.redisService.incr(key);

    // 每10次点击同步到数据库
    const count = await this.redisService.get(key);
    if (parseInt(count) % 10 === 0) {
      await this.prisma.advertisement.update({
        where: { id },
        data: {
          clicks: { increment: parseInt(count) },
        },
      });
      await this.redisService.set(key, '0');
    }
  }

  async getStats(
    userId: bigint,
    startDate?: Date,
    endDate?: Date,
  ): Promise<AdStats> {
    const where: any = { deletedAt: null };

    if (startDate) {
      where.createdAt = { gte: startDate };
    }
    if (endDate) {
      where.createdAt = { ...where.createdAt, lte: endDate };
    }

    const ads = await this.prisma.advertisement.findMany({
      where,
      orderBy: { createdAt: 'desc' },
    });

    // 计算统计数据
    const stats = this.calculateStats(ads);

    return stats;
  }

  private calculateStats(ads: Advertisement[]): AdStats {
    const totalImpressions = ads.reduce((sum, ad) => sum + ad.impressions, 0);
    const totalClicks = ads.reduce((sum, ad) => sum + ad.clicks, 0);
    const ctr = totalImpressions > 0
      ? (totalClicks / totalImpressions * 100).toFixed(2)
      : '0.00';
    const cpmCost = Math.ceil(totalImpressions / 1000) * 0.05;
    const cpcCost = totalClicks * 0.10;
    const totalCost = cpmCost + cpcCost;

    return {
      totalAds: ads.length,
      totalImpressions,
      totalClicks,
      ctr: parseFloat(ctr),
      cpmCost,
      cpcCost,
      totalCost,
      adsByPosition: this.groupByPosition(ads),
    };
  }

  private groupByPosition(ads: Advertisement[]): Record<string, number> {
    return ads.reduce((acc, ad) => {
      acc[ad.position] = (acc[ad.position] || 0) + 1;
      return acc;
    }, {});
  }
```

## 6. 控制器设计

### 6.1 AdvertisementController

```typescript
@Controller('ads')
export class AdvertisementController {
  constructor(private adService: AdvertisementService) {}

  @Get()
  async findAll(@Query('position') position?: string) {
    return this.adService.findAll(position, 'active');
  }

  @Get('stats')
  @UseGuards(AdminAuthGuard)
  async getStats(
    @User() user: User,
    @Query('startDate') startDate?: string,
    @Query('endDate') endDate?: string,
  ) {
    const start = startDate ? new Date(startDate) : undefined;
    const end = endDate ? new Date(endDate) : undefined;
    return this.adService.getStats(BigInt(user.id), start, end);
  }

  @Post()
  @UseGuards(AdminAuthGuard)
  async create(@Body() dto: CreateAdvertisementDto) {
    return this.adService.create(dto);
  }

  @Put(':id')
  @UseGuards(AdminAuthGuard)
  async update(
    @Param('id') id: string,
    @Body() dto: UpdateAdvertisementDto,
  ) {
    return this.adService.update(BigInt(id), dto);
  }

  @Delete(':id')
  @UseGuards(AdminAuthGuard)
  async delete(@Param('id') id: string) {
    return this.adService.delete(BigInt(id));
  }

  @Post(':id/impression')
  async recordImpression(@Param('id') id: string) {
    return this.adService.recordImpression(BigInt(id));
  }

  @Post(':id/click')
  async recordClick(@Param('id') id: string) {
    return this.adService.recordClick(BigInt(id));
  }
}
```

## 7. 前端集成

### 7.1 广告轮播组件

```vue
<template>
  <view class="ad-banner" v-if="currentAd">
    <image :src="currentAd.image" mode="aspectFill" />
    <view class="ad-info">{{ currentAd.title }}</view>
  </view>
</template>

<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue';

const props = defineProps<{
  position: string;
}>();

const currentAd = ref<Advertisement | null>(null);
let timer: NodeJS.Timeout | null = null;

const fetchAds = async () => {
  const response = await api.get('/ads', { position: props.position });
  startRotation(response.data);
};

const startRotation = (ads: Advertisement[]) => {
  if (ads.length === 0) return;

  let index = 0;
  currentAd.value = ads[index];
  recordImpression(ads[index].id);

  timer = setInterval(() => {
    index = (index + 1) % ads.length;
    currentAd.value = ads[index];
    recordImpression(ads[index].id);
  }, 30000); // 30秒轮播
};

const recordImpression = async (adId: string) => {
  await api.post(`/ads/${adId}/impression`);
};

const handleClick = async () => {
  if (currentAd.value) {
    await api.post(`/ads/${currentAd.value.id}/click`);
    window.open(currentAd.value.link, '_blank');
  }
};

onMounted(() => {
  fetchAds();
});

onUnmounted(() => {
  if (timer) clearInterval(timer);
});
</script>
```

```typescript
@Cron('0 * * * *')
async syncAdDataToDatabase() {
  // 同步展示次数
  const impressionKeys = await this.redisService.keys('ad:impression:*');
  for (const key of impressionKeys) {
    const adId = BigInt(key.split(':')[1]);
```

## 10. 性能优化

### 10.1 缓存策略

```typescript
// 缓存可展示广告
const CACHE_KEY = `ads:available:${position}`;
const CACHE_TTL = 300; // 5分钟

async getAvailableAds(position: string): Promise<Advertisement[]> {
  const cached = await this.redisService.get(CACHE_KEY);
  if (cached) {
    return JSON.parse(cached);
  }

  const ads = await this.prisma.advertisement.findMany({
    where: {
      position,
      status: 'active',
      startDate: { lte: new Date() },
      endDate: { gte: new Date() },
    },
  });

  await this.redisService.set(CACHE_KEY, JSON.stringify(ads), CACHE_TTL);
  return ads;
}
```

### 10.2 展示和点击记录优化

- **批量同步**：Redis 累积到一定数量后同步到数据库
- **异步处理**：展示和点击记录异步处理，不影响用户访问
- **去重处理**：同一用户同一广告一定时间内只计费一次

### 10.3 查询优化

- **索引优化**：(position, status, startDate, endDate)
- **只查必要字段**：避免 `SELECT *`
- **分页查询**：使用 cursor-based 分页
